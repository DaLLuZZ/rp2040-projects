165 void startPhaseLabyrinth(){
166 if(!sdCardMalfunctioning){
167 auto l = SDReader::ReadMapsFromSD(currentMapName);
168 auto tuple = Labyrinth::convertStringLab(l);
169 currentLabData = std::get<0>(tuple);
170 highscore = std::get<1>(tuple);
171 }else{
172 currentLabData = Labyrinth::getDefaultLab();
173 highscore = 404;
174 }
175
176 labyrinth->InitLabyrinth(currentLabData, visualizer, mode);
177 p->InitPlayer(labyrinth->getStartCoords()[0],labyrinth->getStartCoords()[1], labyrinth);
178 visualizer->displayBlankScreen();
179 if(mode != FLASH){
180 labyrinth->visualizeLab(p->getX(),p->getY());
181 }else{
182 labyrinth->visualizeLab(p->getX(),p->getY(), true);
183 task::sleep(2000);
184 labyrinth->visualizeLab(p->getX(),p->getY());
185 }
186
187 p->StartPlayerStatCounts();
188 while(currentPhase == 1){
189 if(!bottom_button.gpioRead()) {
190 uint16_t x_value = joy_X.adcReadRaw();
191 uint16_t y_value = joy_Y.adcReadRaw();
192 int x_diff = x_value - x_offset;
193 int y_diff = y_value - y_offset;
194
195 if(p->movePlayer(x_diff,y_diff)){
196 labyrinth->visualizeLab(p->getX(), p->getY());
197 task::sleep(GAMESPEED);
198 }
199 }
200
201 if(p->getX() == labyrinth->getFinishCoords()[0] && p->getY() == labyrinth->getFinishCoords()[1])
202 {
203 currentPhase = 2;
204 }
205 }
206 };
207
209 void startPhaseEnd(){
210 bool highscoreBeaten = false;
211 auto stats = p->getStats();
212 stats[2] *= ((double)1+((double)mode / 10));
213 if(stats[2] > highscore){
214 highscore = stats[2];
215 highscoreBeaten = true;
216 }
217 visualizer->displayEndScreen(stats, highscore, currentMapName);
218
219 if(!sdCardMalfunctioning && highscoreBeaten){
220 SDWriter::WriteLab(currentMapName, currentLabData, highscore);
221 }
222
223 bounceGuardBottom = true;
224 while (currentPhase == 2){
225 if(bottom_button.gpioRead()){
226 bounceGuardBottom = false;
227 task::sleep(DEBOUNCESLEEP);
228 };
229 if(!bottom_button.gpioRead() && !bounceGuardBottom){
230 currentPhase = 0;
231 bounceGuardBottom = true;
232 }
233 }
234 visualizer->displayBlankScreen();
235 };
236};
